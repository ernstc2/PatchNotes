  <action>
1. Create `src/lib/db/schema/system.ts`:
   ```typescript
   import { pgTable, uuid, timestamp, text } from 'drizzle-orm/pg-core';

   export const systemStatus = pgTable('system_status', {
     id: uuid().primaryKey().defaultRandom(),
     service: text().notNull(),
     status: text().notNull(),
     message: text(),
     checkedAt: timestamp({ withTimezone: true }).defaultNow().notNull(),
   });

   export type SystemStatus = typeof systemStatus.$inferSelect;
   export type NewSystemStatus = typeof systemStatus.$inferInsert;
   ```

2. Create `src/lib/db/schema/index.ts`:
   ```typescript
   export * from './system';
   ```

3. Create `src/lib/db/index.ts`:
   ```typescript
   import { neon } from '@neondatabase/serverless';
   import { drizzle } from 'drizzle-orm/neon-http';
   import * as schema from './schema';

   const sql = neon(process.env.DATABASE_URL!);
   export const db = drizzle({ client: sql, schema, casing: 'snake_case' });
   ```

4. Create `src/app/api/health/route.ts`:
   ```typescript
   import { db } from '@/lib/db';
   import { systemStatus } from '@/lib/db/schema';

   export async function GET() {
     try {
       await db.select().from(systemStatus).limit(1);
       return Response.json({ status: 'ok', timestamp: new Date().toISOString() });
     } catch (error) {
       return Response.json(
         { status: 'error', message: error instanceof Error ? error.message : 'Unknown error' },
         { status: 500 }
       );
     }
   }
   ```

5. Run `npx tsc --noEmit` to confirm all types resolve correctly.

Note: Do NOT run `drizzle-kit generate` or `drizzle-kit migrate` here — that requires a live DATABASE_URL connection which is a user setup concern. The schema files and config are ready for when the user connects their Neon database.
